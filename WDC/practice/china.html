Web端数据可视化的研究和应用

6.1.1 简介

地图图表是数据可视化的重要组成部分，传统的Web地图图表，没有时间上的维度。但有些数据必须通过时间的变化才能更好的体现其意义。

本实例是基于Acharts的基础类库，开发的地图区域Web可视化应用。全例有4个抽象类，共1336行（含注释）。5套数据，data.js、dataHumidity、dataRain、dataSunShine、dataTemperature，其中data.js是地图数据，来源于网络。后四者是信息数据源，来源于《中国统计年鉴》。

6.1.2 绘制基础地图

map.js 和 region.js 负责实现基础地图。其中map.js为主类，负责调度数据、绑定时间、实现主要逻辑，统筹整个图表。region.js 是区域绘制类，负责绘制区域图形。相对于map.js而言region.js的抽象程度更高，可适用于所有树型数据结构的区域绘制。map.js共有6组可配信息，其中由于Acharts的绘图类库是Raphael，所以attr可参考Raphael的官方文档进行配置。


图6.1 

seajs.use(['acharts','../index'], function(Acharts,Map) { 
    var Series = AChart.Series; 
    Series.Map = Map;

    var chart = new AChart({
        id : 'c1',
        width : 800,
        height : 800,
        legend : null,
        tooltip : null,
        series : [
          {   
            type : 'Map',
            /**
            * 选择地图显示类型
            * @type {string} distribute,simple
            */
            mapName : '中国', //选择地图
            cfg : {
                /**
                * 选择地图显示类型
                * @type {string} distribute,simple
                */
                type : 'simple',//简单地图
                /**
                * 配置区域路径
                * @type {attr}
                */    
                path : {
                    stroke : 'red',
                    fill : '',
                },
                /**
                * 配置中心点
                * @type {attr}
                */
                centerPoint : {
                    fill : 'red'
                },
                /**
                * 配置名称
                * @type {attr}
                */
                text : {
                    fill : 'black'
                },
                /**
                * 是否显示省府地区圆点
                * @type {Boolean}
                */
                show : {
                    centerCircle : true,
                    name : true
                }
            }
          }
        ]
    });
    chart.render();
});

6.1.3 绘制数据分布地图

单纯的简单地图不能直观的显示数据，我们需要找个直观，辨识度高的视觉参量来显示数据。根据生活经验，很直觉的我们会采用颜色的差别来体现数据之间的关系。计算机中，显示颜色通常使用RGB，但这三个数值并不能很好的反应人眼的视觉特点，此时我们需要采用另一套色彩系统——HSL。本例就是通过，改变其中的S，从而反应其数据的变化。

seajs.use(['acharts','../index','../src/dataTemperature'], function(Acharts,Map,Datas) { 
    var Series = AChart.Series;
        Series.Map = Map;
    var chart = new AChart({
        id : 'c1',
        width : 800,
        height : 800,
        legend : null,
        tooltip : {
          title : {
            'font-size' : '16px',
            y : 10,
            x : 5,
            fill : '#ff7c26'
          }
        },
        series : [
          {   
              name : '气温',
            type : 'Map',
            mapName : '中国', //选择地图
            cfg : {
                type : 'distribute',//数据分布
                path : {
                    stroke : '',
                    fill : 'red',
                },
                text : {
                    fill : 'black'
                },
                show : {
                    centerCircle : false,
                    name : true
                }
            },
            timeline : {
                itemCfg : {
                        color : '#ff6600',
                        type : 'circle'    
                },
                timeNode : Datas.timeNode,
                dy : 30
            },
            datas: Datas.datas
          }
        ]
    });

    chart.render();


});
